Customers Customer Rest Controller
/api/customers
GET
/all Returns all customer information available
GET
/by-id Returns customer information by ID [param: id (Integer)]
GET
/by-name Returns customer information by name [param: name (String)]
GET
/by-phone Returns customer information by phone [param: phone (String)]
GET
/by-email Returns customer information by email [param: email (String)]
POST
/new Adds new customer to the database and returns [body: Customer]
PATCH
/update Updates customer information by id and returns [param: id, body: Customer]
Products Product Rest Controller
/api/products
GET

GET

GET

GET

PATCH

Orders Order Rest Controller
/api/orders
GET
/all Returns all order information available
GET
/by-order-id Returns order information by order ID [param: order-id (Integer)]
GET
/by-customer-id Returns order information by customer ID [param: customer-id (Integer)]
GET
/by-product-id Returns order information by product ID [param: product-id (Integer)]
GET
/by-paid Returns order information by paid status [param: paid (Boolean)]
GET
/by-placed-on Returns order information on date [param: date (Date)]
GET
/by-placed-month Returns order information in month [param: month (Integer)]
GET
/by-placed-year Returns order information in year [param: year (Integer)]
GET
/by-price-range Returns order information within price range [param: min, max (Double)]
POST
/new Adds new order information to the database and returns [body: Order]
PATCH
/update Updates order information by id and returns [param: id, body: Order]
DELETE
/delete Deletes order information by id [param: id]
